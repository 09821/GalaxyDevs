local function makeDraggable(frame, dragHandle)
    local dragging, dragInput, dragStart, startPos
    
    dragHandle = dragHandle or frame
    
    dragHandle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)
    
    RunService.Heartbeat:Connect(function()
        if dragging and dragInput then
            local delta = dragInput.Position - dragStart
            frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        end
    end)
end

local function getObjectPosition(obj)
    if obj:IsA("Tool") then
        local handle = obj:FindFirstChild("Handle")
        return handle and handle.Position or nil
    elseif obj:IsA("Model") then
        local primaryPart = obj.PrimaryPart or obj:FindFirstChildWhichIsA("BasePart")
        return primaryPart and primaryPart.Position or nil
    end
    return nil
end

local function createESP(obj, displayName, rarity)
    if not CONFIG.EspEnabled then return end
    
    local position = getObjectPosition(obj)
    if not position then return end
    
    if espObjects[obj] then
        espObjects[obj].billboard:Destroy()
    end
    
    local billboard = Instance.new("BillboardGui")
    billboard.Name = "FruitESP"
    billboard.Adornee = obj:IsA("Tool") and obj:FindFirstChild("Handle") or obj.PrimaryPart or obj:FindFirstChildWhichIsA("BasePart")

    billboard.AlwaysOnTop = true

local function detectFruits()
    detectedFruits = {}
    for _, obj in ipairs(game.Workspace:GetDescendants()) do
        if (obj:IsA("Tool") or obj:IsA("Model")) and (validFruitsDict[obj.Name] or obj.Name == "Fruit") then
            table.insert(detectedFruits, obj)
        end
    end
    return detectedFruits
end

local function checkForWhitelistedFruits()
    if #CONFIG.WhitelistFruits == 0 then
        return false
    end
    
    detectFruits()
    
    for _, fruit in ipairs(detectedFruits) do
        for _, whitelistName in ipairs(CONFIG.WhitelistFruits) do
            if fruit.Name:lower():find(whitelistName:lower()) then
                return true, fruit
            end
        end
    end
    
    return false
end

local function collectFruit(fruit)

local function autoAfkLogic()
    if not CONFIG.AutoAfkEnabled then return end
    
    CONFIG.ServersChecked = CONFIG.ServersChecked + 1
    
    local found, fruit = checkForWhitelistedFruits()
    
    if found then
        if CONFIG.ServersChecked % 3 == 0 then
            -- A cada 3 servidores com frutas, coleta ao invÃ©s de notificar
            collectFruit(fruit)
            notify("Auto AFK", "Fruta coletada! Aguardando movimento...")
        else
            -- Notifica no Discord
            detectFruits()
            local fruitsToNotify = {}
            for _, f in ipairs(detectedFruits) do
                for _, wl in ipairs(CONFIG.WhitelistFruits) do
                    if f.Name:lower():find(wl:lower()) then
                        table.insert(fruitsToNotify, f)
                        break
                    end
                end
            end
            sendDiscordWebhook(fruitsToNotify)
            
            -- Vai para prÃ³ximo servidor
            task.wait(3)
            local servers = game:GetService("HttpService"):JSONDecode(game:HttpGet("https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"))
            if servers and servers.data and #servers.data > 0 then
                TeleportService:TeleportToPlaceInstance(game.PlaceId, servers.data[1].id, LocalPlayer)
            end
        end
    else
        if CONFIG.AutoTpServerWhitelist then
            task.wait(2)
            local servers = game:GetService("HttpService"):JSONDecode(game:HttpGet("https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"))
            if servers and servers.data and #servers.data > 0 then
                TeleportService:TeleportToPlaceInstance(game.PlaceId, servers.data[1].id, LocalPlayer)
            end
        end
    end
end
